3.   hardwares database
tables:

CUSTOMER
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  goods - textfield(min2  max10,blank=false,null=false)

EMPLOYERS
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  salary - currency(min0 max=30,000)
  hours - number(max8)

SUPPLIER
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  sales - charfield(min4 max20)

GOODS
  quality - charfield((min4 max10))
  market price - currency(BigDecimal)
  quantity - number(min10 max10000)

TRANSPORTATION
  transporter - charfield(min4 max20)
  branch - charfield(min4 max40)
  price - currency(BigDecimal)

RELATIONSHIPS

CUSTOMER - EMPLOYERS(M-M)
CUSTOMER - GOODS(M-M)
TRANSPORTATION - CUSTOMER(1-M)
TRANSPORTATION - GOODS(1-M)
SUPPLIER - GOODS(1-M)
SUPPLIER - TRANSPORTATION(1-1)
