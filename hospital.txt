1.   hospital database
tables:

PATIENT
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  diseases - charfield(min5 max40)
  age - autonumber(min0 max=150)
  phone - number(max10)
  address - charfield(min4 max20)
  patient id - number(unique, PK)
  medicine - charfield(min5 max40)
  allergies - charfield(min5 max40)

DOCTOR
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  doctor_id - number(unique)
  phone - number(max10)
  experience - charfield(max3)
  specialist - charfield(max7)
  salary - currency(min0 max=300,000)
  hours - number(max8)

APPOINTMENT
  date - 
  doctor - charfield(min3 max15)
  appoint_id - number(unique, PK)
  doctor_id - number(unique, FK)
  patient id - number(FK)
  blood pressure - charfield
  pulse - charfield(min4 max10)
  treatment - charfield(min5 max50)

VISITOR
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  patient - charfield(min2  max10,blank=false,null=false)
  hours - number(max8)

STAFF
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  salary - currency(min0 max=30,000)
  hours - number(max8)

WORKERS
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  salary - currency(min0 max=30,000)
  hours - number(max8)

RELATIONSHIPS

PATIENT - VISITOR(1-M)
DOCTOR - PATIENT(1-1)
DOCTOR - DOCTOR(1-M)
APPOINTMENT - DOCTOR(1-1)
APPOINTMENT - PATIENT(1-1)
