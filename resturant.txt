4.   resturant database
tables:

CUSTOMER
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  items - textfield(min1 max100)

SUPPLIER
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  stock - charfield(min5 max20,PK)

CHEF
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  chef_id - number(unique, PK)
  type of food - charfield
  salary - currency(BigDecimal)
  experience - charfield(max2)

WAITER
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  orders - charfield(min0 max 20,required)
  salary - currency(BigDecimal)

GOODS
  quality - charfield(max10)
  market price - currency(BigDecimal)
  quantity - number(min50, PK)
  stock - charfield(min5 max20,FK)

MEAL 
  name - charfield(min4 max15)
  meal_id - charfield(unique, PK)
  price - currency(BigDecimal)
  chef_id - number(FK)


DELIVERY
  transporter - charfield(min5 max20)
  branch - charfield(min5 max20)
  price - currency(BigDecimal)

WORKERS
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  salary - currency(BigDecimal)
  hours - number(max8)

RELATIONSHIPS

CUSTOMER - WAITER(M-M)
DELIVERY- CUSTOMER(1-M)
CHEF - WAITER(1-1)
SUPPLIER - GOODS(M-M)
CHEF - MEAL(1-M)
