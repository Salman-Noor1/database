5.   logistics database
tables:

IMPORTER
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  items - textfield(required)

CLIENT
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  client_id - number(unique, PK)
  sales - charfield(required)

STOCK
  item_id - number(unique, PK)
  name - charfield(min5 max50, blank=false)
  quantity - number(required)
  price - currency(BigDecimal)
SALES
  

EMPLOYERS
  name - charfield(min2  max10,blank=false,null=false)
  gender - bool/charfield(required)
  age - autonumber(min0 max=150)
  salary - currency(min0 max=30,000)
  hours - number(max8)
  
TRANSPORTATION
  transporter - charfield(min5 max40)
  branch - charfield(min5 max30)
  cost - currency(BigDecimal)

RELATIONSHIPS

CLIENT - EMPLOYERS(M-M)
IMPORTER - CLIENT(1-M)
TRANSPORTATION - EMPLOYERS(1-M)
EMPLOYERS - IMPORTER(M-M)
